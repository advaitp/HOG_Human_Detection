\DoxyCodeLine{36 \textcolor{comment}{// Implements class templates NiceMock, NaggyMock, and StrictMock.}}
\DoxyCodeLine{37 \textcolor{comment}{//}}
\DoxyCodeLine{38 \textcolor{comment}{// Given a mock class MockFoo that is created using Google Mock,}}
\DoxyCodeLine{39 \textcolor{comment}{// NiceMock<MockFoo> is a subclass of MockFoo that allows}}
\DoxyCodeLine{40 \textcolor{comment}{// uninteresting calls (i.e. calls to mock methods that have no}}
\DoxyCodeLine{41 \textcolor{comment}{// EXPECT\_CALL specs), NaggyMock<MockFoo> is a subclass of MockFoo}}
\DoxyCodeLine{42 \textcolor{comment}{// that prints a warning when an uninteresting call occurs, and}}
\DoxyCodeLine{43 \textcolor{comment}{// StrictMock<MockFoo> is a subclass of MockFoo that treats all}}
\DoxyCodeLine{44 \textcolor{comment}{// uninteresting calls as errors.}}
\DoxyCodeLine{45 \textcolor{comment}{//}}
\DoxyCodeLine{46 \textcolor{comment}{// Currently a mock is naggy by default, so MockFoo and}}
\DoxyCodeLine{47 \textcolor{comment}{// NaggyMock<MockFoo> behave like the same.  However, we will soon}}
\DoxyCodeLine{48 \textcolor{comment}{// switch the default behavior of mocks to be nice, as that in general}}
\DoxyCodeLine{49 \textcolor{comment}{// leads to more maintainable tests.  When that happens, MockFoo will}}
\DoxyCodeLine{50 \textcolor{comment}{// stop behaving like NaggyMock<MockFoo> and start behaving like}}
\DoxyCodeLine{51 \textcolor{comment}{// NiceMock<MockFoo>.}}
\DoxyCodeLine{52 \textcolor{comment}{//}}
\DoxyCodeLine{53 \textcolor{comment}{// NiceMock, NaggyMock, and StrictMock "{}inherit"{} the constructors of}}
\DoxyCodeLine{54 \textcolor{comment}{// their respective base class, with up-\/to 10 arguments.  Therefore}}
\DoxyCodeLine{55 \textcolor{comment}{// you can write NiceMock<MockFoo>(5, "{}a"{}) to construct a nice mock}}
\DoxyCodeLine{56 \textcolor{comment}{// where MockFoo has a constructor that accepts (int, const char*),}}
\DoxyCodeLine{57 \textcolor{comment}{// for example.}}
\DoxyCodeLine{58 \textcolor{comment}{//}}
\DoxyCodeLine{59 \textcolor{comment}{// A known limitation is that NiceMock<MockFoo>, NaggyMock<MockFoo>,}}
\DoxyCodeLine{60 \textcolor{comment}{// and StrictMock<MockFoo> only works for mock methods defined using}}
\DoxyCodeLine{61 \textcolor{comment}{// the MOCK\_METHOD* family of macros DIRECTLY in the MockFoo class.}}
\DoxyCodeLine{62 \textcolor{comment}{// If a mock method is defined in a base class of MockFoo, the "{}nice"{}}}
\DoxyCodeLine{63 \textcolor{comment}{// or "{}strict"{} modifier may not affect it, depending on the compiler.}}
\DoxyCodeLine{64 \textcolor{comment}{// In particular, nesting NiceMock, NaggyMock, and StrictMock is NOT}}
\DoxyCodeLine{65 \textcolor{comment}{// supported.}}
\DoxyCodeLine{66 \textcolor{comment}{//}}
\DoxyCodeLine{67 \textcolor{comment}{// Another known limitation is that the constructors of the base mock}}
\DoxyCodeLine{68 \textcolor{comment}{// cannot have arguments passed by non-\/const reference, which are}}
