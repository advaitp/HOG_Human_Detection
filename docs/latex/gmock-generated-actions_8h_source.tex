\DoxyCodeLine{784 \textcolor{comment}{// The ACTION* family of macros can be used in a namespace scope to}}
\DoxyCodeLine{785 \textcolor{comment}{// define custom actions easily.  The syntax:}}
\DoxyCodeLine{786 \textcolor{comment}{//}}
\DoxyCodeLine{787 \textcolor{comment}{//   ACTION(name) \{ statements; \}}}
\DoxyCodeLine{788 \textcolor{comment}{//}}
\DoxyCodeLine{789 \textcolor{comment}{// will define an action with the given name that executes the}}
\DoxyCodeLine{790 \textcolor{comment}{// statements.  The value returned by the statements will be used as}}
\DoxyCodeLine{791 \textcolor{comment}{// the return value of the action.  Inside the statements, you can}}
\DoxyCodeLine{792 \textcolor{comment}{// refer to the K-\/th (0-\/based) argument of the mock function by}}
\DoxyCodeLine{793 \textcolor{comment}{// 'argK', and refer to its type by 'argK\_type'.  For example:}}
\DoxyCodeLine{794 \textcolor{comment}{//}}
\DoxyCodeLine{795 \textcolor{comment}{//   ACTION(IncrementArg1) \{}}
\DoxyCodeLine{796 \textcolor{comment}{//     arg1\_type temp = arg1;}}
\DoxyCodeLine{797 \textcolor{comment}{//     return ++(*temp);}}
\DoxyCodeLine{798 \textcolor{comment}{//   \}}}
\DoxyCodeLine{799 \textcolor{comment}{//}}
\DoxyCodeLine{800 \textcolor{comment}{// allows you to write}}
\DoxyCodeLine{801 \textcolor{comment}{//}}
\DoxyCodeLine{802 \textcolor{comment}{//   ...WillOnce(IncrementArg1());}}
\DoxyCodeLine{803 \textcolor{comment}{//}}
\DoxyCodeLine{804 \textcolor{comment}{// You can also refer to the entire argument tuple and its type by}}
\DoxyCodeLine{805 \textcolor{comment}{// 'args' and 'args\_type', and refer to the mock function type and its}}
\DoxyCodeLine{806 \textcolor{comment}{// return type by 'function\_type' and 'return\_type'.}}
\DoxyCodeLine{807 \textcolor{comment}{//}}
\DoxyCodeLine{808 \textcolor{comment}{// Note that you don't need to specify the types of the mock function}}
\DoxyCodeLine{809 \textcolor{comment}{// arguments.  However rest assured that your code is still type-\/safe:}}
\DoxyCodeLine{810 \textcolor{comment}{// you'll get a compiler error if *arg1 doesn't support the ++}}
\DoxyCodeLine{811 \textcolor{comment}{// operator, or if the type of ++(*arg1) isn't compatible with the}}
\DoxyCodeLine{812 \textcolor{comment}{// mock function's return type, for example.}}
\DoxyCodeLine{813 \textcolor{comment}{//}}
\DoxyCodeLine{814 \textcolor{comment}{// Sometimes you'll want to parameterize the action.   For that you can use}}
\DoxyCodeLine{815 \textcolor{comment}{// another macro:}}

\DoxyCodeLine{820 \textcolor{comment}{//}}
\DoxyCodeLine{821 \textcolor{comment}{//   ACTION\_P(Add, n) \{ return arg0 + n; \}}}
\DoxyCodeLine{822 \textcolor{comment}{//}}
\DoxyCodeLine{823 \textcolor{comment}{// will allow you to write:}}
\DoxyCodeLine{824 \textcolor{comment}{//}}
\DoxyCodeLine{825 \textcolor{comment}{//   ...WillOnce(Add(5));}}
\DoxyCodeLine{826 \textcolor{comment}{//}}
\DoxyCodeLine{827 \textcolor{comment}{// Note that you don't need to provide the type of the parameter}}
\DoxyCodeLine{828 \textcolor{comment}{// either.  If you need to reference the type of a parameter named}}
\DoxyCodeLine{829 \textcolor{comment}{// 'foo', you can write 'foo\_type'.  For example, in the body of}}
\DoxyCodeLine{830 \textcolor{comment}{// ACTION\_P(Add, n) above, you can write 'n\_type' to refer to the type}}
\DoxyCodeLine{831 \textcolor{comment}{// of 'n'.}}
\DoxyCodeLine{832 \textcolor{comment}{//}}
\DoxyCodeLine{833 \textcolor{comment}{// We also provide ACTION\_P2, ACTION\_P3, ..., up to ACTION\_P10 to support}}
\DoxyCodeLine{834 \textcolor{comment}{// multi-\/parameter actions.}}
\DoxyCodeLine{835 \textcolor{comment}{//}}
\DoxyCodeLine{836 \textcolor{comment}{// For the purpose of typing, you can view}}

\DoxyCodeLine{851 \textcolor{comment}{//}}
\DoxyCodeLine{852 \textcolor{comment}{// You can also overload actions with different numbers of parameters:}}
\DoxyCodeLine{853 \textcolor{comment}{//}}

\DoxyCodeLine{895 \textcolor{comment}{// that cannot be inferred from its value parameters.  ACTION() and}}
\DoxyCodeLine{896 \textcolor{comment}{// ACTION\_P*() don't support that.  ACTION\_TEMPLATE() remedies that}}
\DoxyCodeLine{897 \textcolor{comment}{// and can be viewed as an extension to ACTION() and ACTION\_P*().}}
\DoxyCodeLine{898 \textcolor{comment}{//}}
\DoxyCodeLine{899 \textcolor{comment}{// The syntax:}}
\DoxyCodeLine{900 \textcolor{comment}{//}}
\DoxyCodeLine{901 \textcolor{comment}{//   ACTION\_TEMPLATE(ActionName,}}
\DoxyCodeLine{902 \textcolor{comment}{//                   HAS\_m\_TEMPLATE\_PARAMS(kind1, name1, ..., kind\_m, name\_m),}}
\DoxyCodeLine{903 \textcolor{comment}{//                   AND\_n\_VALUE\_PARAMS(p1, ..., p\_n)) \{ statements; \}}}
\DoxyCodeLine{904 \textcolor{comment}{//}}
\DoxyCodeLine{905 \textcolor{comment}{// defines an action template that takes m explicit template}}
\DoxyCodeLine{906 \textcolor{comment}{// parameters and n value parameters.  name\_i is the name of the i-\/th}}
\DoxyCodeLine{907 \textcolor{comment}{// template parameter, and kind\_i specifies whether it's a typename,}}
\DoxyCodeLine{908 \textcolor{comment}{// an integral constant, or a template.  p\_i is the name of the i-\/th}}
\DoxyCodeLine{909 \textcolor{comment}{// value parameter.}}
\DoxyCodeLine{910 \textcolor{comment}{//}}
\DoxyCodeLine{911 \textcolor{comment}{// Example:}}
\DoxyCodeLine{912 \textcolor{comment}{//}}
\DoxyCodeLine{913 \textcolor{comment}{//   // DuplicateArg<k, T>(output) converts the k-\/th argument of the mock}}
\DoxyCodeLine{914 \textcolor{comment}{//   // function to type T and copies it to *output.}}
\DoxyCodeLine{915 \textcolor{comment}{//   ACTION\_TEMPLATE(DuplicateArg,}}
\DoxyCodeLine{916 \textcolor{comment}{//                   HAS\_2\_TEMPLATE\_PARAMS(int, k, typename, T),}}
\DoxyCodeLine{917 \textcolor{comment}{//                   AND\_1\_VALUE\_PARAMS(output)) \{}}
\DoxyCodeLine{918 \textcolor{comment}{//     *output = T(::testing::get<k>(args));}}
\DoxyCodeLine{919 \textcolor{comment}{//   \}}}
\DoxyCodeLine{920 \textcolor{comment}{//   ...}}
\DoxyCodeLine{921 \textcolor{comment}{//     int n;}}
\DoxyCodeLine{922 \textcolor{comment}{//     EXPECT\_CALL(mock, Foo(\_, \_))}}
\DoxyCodeLine{923 \textcolor{comment}{//         .WillOnce(DuplicateArg<1, unsigned char>(\&n));}}
\DoxyCodeLine{924 \textcolor{comment}{//}}
\DoxyCodeLine{925 \textcolor{comment}{// To create an instance of an action template, write:}}

\DoxyCodeLine{2101 \textcolor{comment}{//}}
\DoxyCodeLine{2102 \textcolor{comment}{//   1. The arguments are passed by value by default.  If you need to}}
\DoxyCodeLine{2103 \textcolor{comment}{//   pass an argument by reference, wrap it inside ByRef().  For}}
\DoxyCodeLine{2104 \textcolor{comment}{//   example,}}
\DoxyCodeLine{2105 \textcolor{comment}{//}}
\DoxyCodeLine{2106 \textcolor{comment}{//     InvokeArgument<1>(5, string("{}Hello"{}), ByRef(foo))}}
\DoxyCodeLine{2107 \textcolor{comment}{//}}
\DoxyCodeLine{2108 \textcolor{comment}{//   passes 5 and string("{}Hello"{}) by value, and passes foo by}}
\DoxyCodeLine{2109 \textcolor{comment}{//   reference.}}
\DoxyCodeLine{2110 \textcolor{comment}{//}}
\DoxyCodeLine{2111 \textcolor{comment}{//   2. If the callable takes an argument by reference but ByRef() is}}
\DoxyCodeLine{2112 \textcolor{comment}{//   not used, it will receive the reference to a copy of the value,}}
\DoxyCodeLine{2113 \textcolor{comment}{//   instead of the original value.  For example, when the 0-\/th}}
\DoxyCodeLine{2114 \textcolor{comment}{//   argument of the mock function takes a const string\&, the action}}
\DoxyCodeLine{2115 \textcolor{comment}{//}}
\DoxyCodeLine{2116 \textcolor{comment}{//     InvokeArgument<0>(string("{}Hello"{}))}}
\DoxyCodeLine{2117 \textcolor{comment}{//}}
\DoxyCodeLine{2118 \textcolor{comment}{//   makes a copy of the temporary string("{}Hello"{}) object and passes a}}
\DoxyCodeLine{2119 \textcolor{comment}{//   reference of the copy, instead of the original temporary object,}}
\DoxyCodeLine{2120 \textcolor{comment}{//   to the callable.  This makes it easy for a user to define an}}
\DoxyCodeLine{2121 \textcolor{comment}{//   InvokeArgument action from temporary values and have it performed}}
\DoxyCodeLine{2122 \textcolor{comment}{//   later.}}
