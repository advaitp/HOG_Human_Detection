\DoxyCodeLine{1168 \textcolor{comment}{//}}
\DoxyCodeLine{1169 \textcolor{comment}{// Basic Usage}}
\DoxyCodeLine{1170 \textcolor{comment}{// ===========}}
\DoxyCodeLine{1171 \textcolor{comment}{//}}
\DoxyCodeLine{1172 \textcolor{comment}{// The syntax}}
\DoxyCodeLine{1173 \textcolor{comment}{//}}
\DoxyCodeLine{1174 \textcolor{comment}{//   MATCHER(name, description\_string) \{ statements; \}}}
\DoxyCodeLine{1175 \textcolor{comment}{//}}
\DoxyCodeLine{1176 \textcolor{comment}{// defines a matcher with the given name that executes the statements,}}
\DoxyCodeLine{1177 \textcolor{comment}{// which must return a bool to indicate if the match succeeds.  Inside}}
\DoxyCodeLine{1178 \textcolor{comment}{// the statements, you can refer to the value being matched by 'arg',}}
\DoxyCodeLine{1179 \textcolor{comment}{// and refer to its type by 'arg\_type'.}}
\DoxyCodeLine{1180 \textcolor{comment}{//}}
\DoxyCodeLine{1181 \textcolor{comment}{// The description string documents what the matcher does, and is used}}
\DoxyCodeLine{1182 \textcolor{comment}{// to generate the failure message when the match fails.  Since a}}
\DoxyCodeLine{1183 \textcolor{comment}{// MATCHER() is usually defined in a header file shared by multiple}}
\DoxyCodeLine{1184 \textcolor{comment}{// C++ source files, we require the description to be a C-\/string}}
\DoxyCodeLine{1185 \textcolor{comment}{// literal to avoid possible side effects.  It can be empty, in which}}
\DoxyCodeLine{1186 \textcolor{comment}{// case we'll use the sequence of words in the matcher name as the}}
\DoxyCodeLine{1187 \textcolor{comment}{// description.}}
\DoxyCodeLine{1188 \textcolor{comment}{//}}
\DoxyCodeLine{1189 \textcolor{comment}{// For example:}}
\DoxyCodeLine{1190 \textcolor{comment}{//}}
\DoxyCodeLine{1191 \textcolor{comment}{//   MATCHER(IsEven, "{}"{}) \{ return (arg \% 2) == 0; \}}}
\DoxyCodeLine{1192 \textcolor{comment}{//}}
\DoxyCodeLine{1193 \textcolor{comment}{// allows you to write}}
\DoxyCodeLine{1194 \textcolor{comment}{//}}
\DoxyCodeLine{1195 \textcolor{comment}{//   // Expects mock\_foo.Bar(n) to be called where n is even.}}
\DoxyCodeLine{1196 \textcolor{comment}{//   EXPECT\_CALL(mock\_foo, Bar(IsEven()));}}
\DoxyCodeLine{1197 \textcolor{comment}{//}}
\DoxyCodeLine{1198 \textcolor{comment}{// or,}}
\DoxyCodeLine{1199 \textcolor{comment}{//}}
\DoxyCodeLine{1200 \textcolor{comment}{//   // Verifies that the value of some\_expression is even.}}
\DoxyCodeLine{1201 \textcolor{comment}{//   EXPECT\_THAT(some\_expression, IsEven());}}
\DoxyCodeLine{1202 \textcolor{comment}{//}}
\DoxyCodeLine{1203 \textcolor{comment}{// If the above assertion fails, it will print something like:}}
\DoxyCodeLine{1204 \textcolor{comment}{//}}
\DoxyCodeLine{1205 \textcolor{comment}{//   Value of: some\_expression}}
\DoxyCodeLine{1206 \textcolor{comment}{//   Expected: is even}}
\DoxyCodeLine{1207 \textcolor{comment}{//     Actual: 7}}
\DoxyCodeLine{1208 \textcolor{comment}{//}}
\DoxyCodeLine{1209 \textcolor{comment}{// where the description "{}is even"{} is automatically calculated from the}}
\DoxyCodeLine{1210 \textcolor{comment}{// matcher name IsEven.}}
\DoxyCodeLine{1211 \textcolor{comment}{//}}
\DoxyCodeLine{1212 \textcolor{comment}{// Argument Type}}
\DoxyCodeLine{1213 \textcolor{comment}{// =============}}
\DoxyCodeLine{1214 \textcolor{comment}{//}}
\DoxyCodeLine{1215 \textcolor{comment}{// Note that the type of the value being matched (arg\_type) is}}
\DoxyCodeLine{1216 \textcolor{comment}{// determined by the context in which you use the matcher and is}}
\DoxyCodeLine{1217 \textcolor{comment}{// supplied to you by the compiler, so you don't need to worry about}}
\DoxyCodeLine{1218 \textcolor{comment}{// declaring it (nor can you).  This allows the matcher to be}}
\DoxyCodeLine{1219 \textcolor{comment}{// polymorphic.  For example, IsEven() can be used to match any type}}
\DoxyCodeLine{1220 \textcolor{comment}{// where the value of "{}(arg \% 2) == 0"{} can be implicitly converted to}}
\DoxyCodeLine{1221 \textcolor{comment}{// a bool.  In the "{}Bar(IsEven())"{} example above, if method Bar()}}
\DoxyCodeLine{1222 \textcolor{comment}{// takes an int, 'arg\_type' will be int; if it takes an unsigned long,}}
\DoxyCodeLine{1223 \textcolor{comment}{// 'arg\_type' will be unsigned long; and so on.}}
\DoxyCodeLine{1224 \textcolor{comment}{//}}
\DoxyCodeLine{1225 \textcolor{comment}{// Parameterizing Matchers}}
\DoxyCodeLine{1226 \textcolor{comment}{// =======================}}
\DoxyCodeLine{1227 \textcolor{comment}{//}}
\DoxyCodeLine{1228 \textcolor{comment}{// Sometimes you'll want to parameterize the matcher.  For that you}}
\DoxyCodeLine{1229 \textcolor{comment}{// can use another macro:}}
\DoxyCodeLine{1230 \textcolor{comment}{//}}
\DoxyCodeLine{1231 \textcolor{comment}{//   MATCHER\_P(name, param\_name, description\_string) \{ statements; \}}}
\DoxyCodeLine{1232 \textcolor{comment}{//}}
\DoxyCodeLine{1233 \textcolor{comment}{// For example:}}
\DoxyCodeLine{1234 \textcolor{comment}{//}}
\DoxyCodeLine{1235 \textcolor{comment}{//   MATCHER\_P(HasAbsoluteValue, value, "{}"{}) \{ return abs(arg) == value; \}}}
\DoxyCodeLine{1236 \textcolor{comment}{//}}
\DoxyCodeLine{1237 \textcolor{comment}{// will allow you to write:}}
\DoxyCodeLine{1238 \textcolor{comment}{//}}
\DoxyCodeLine{1239 \textcolor{comment}{//   EXPECT\_THAT(Blah("{}a"{}), HasAbsoluteValue(n));}}
\DoxyCodeLine{1240 \textcolor{comment}{//}}
\DoxyCodeLine{1241 \textcolor{comment}{// which may lead to this message (assuming n is 10):}}
\DoxyCodeLine{1242 \textcolor{comment}{//}}
\DoxyCodeLine{1243 \textcolor{comment}{//   Value of: Blah("{}a"{})}}
\DoxyCodeLine{1244 \textcolor{comment}{//   Expected: has absolute value 10}}
\DoxyCodeLine{1245 \textcolor{comment}{//     Actual: -\/9}}
\DoxyCodeLine{1246 \textcolor{comment}{//}}
\DoxyCodeLine{1247 \textcolor{comment}{// Note that both the matcher description and its parameter are}}
\DoxyCodeLine{1248 \textcolor{comment}{// printed, making the message human-\/friendly.}}
\DoxyCodeLine{1249 \textcolor{comment}{//}}
\DoxyCodeLine{1250 \textcolor{comment}{// In the matcher definition body, you can write 'foo\_type' to}}
\DoxyCodeLine{1251 \textcolor{comment}{// reference the type of a parameter named 'foo'.  For example, in the}}
\DoxyCodeLine{1252 \textcolor{comment}{// body of MATCHER\_P(HasAbsoluteValue, value) above, you can write}}
\DoxyCodeLine{1253 \textcolor{comment}{// 'value\_type' to refer to the type of 'value'.}}
\DoxyCodeLine{1254 \textcolor{comment}{//}}
\DoxyCodeLine{1255 \textcolor{comment}{// We also provide MATCHER\_P2, MATCHER\_P3, ..., up to MATCHER\_P10 to}}
\DoxyCodeLine{1256 \textcolor{comment}{// support multi-\/parameter matchers.}}
\DoxyCodeLine{1257 \textcolor{comment}{//}}
\DoxyCodeLine{1258 \textcolor{comment}{// Describing Parameterized Matchers}}
\DoxyCodeLine{1259 \textcolor{comment}{// =================================}}
\DoxyCodeLine{1260 \textcolor{comment}{//}}
\DoxyCodeLine{1261 \textcolor{comment}{// The last argument to MATCHER*() is a string-\/typed expression.  The}}
\DoxyCodeLine{1262 \textcolor{comment}{// expression can reference all of the matcher's parameters and a}}
\DoxyCodeLine{1263 \textcolor{comment}{// special bool-\/typed variable named 'negation'.  When 'negation' is}}
\DoxyCodeLine{1264 \textcolor{comment}{// false, the expression should evaluate to the matcher's description;}}
\DoxyCodeLine{1265 \textcolor{comment}{// otherwise it should evaluate to the description of the negation of}}
\DoxyCodeLine{1266 \textcolor{comment}{// the matcher.  For example,}}
\DoxyCodeLine{1267 \textcolor{comment}{//}}
\DoxyCodeLine{1268 \textcolor{comment}{//   using testing::PrintToString;}}
\DoxyCodeLine{1269 \textcolor{comment}{//}}
\DoxyCodeLine{1270 \textcolor{comment}{//   MATCHER\_P2(InClosedRange, low, hi,}}
\DoxyCodeLine{1271 \textcolor{comment}{//       string(negation ? "{}is not"{} : "{}is"{}) + "{} in range ["{} +}}
\DoxyCodeLine{1272 \textcolor{comment}{//       PrintToString(low) + "{}, "{} + PrintToString(hi) + "{}]"{}) \{}}
\DoxyCodeLine{1273 \textcolor{comment}{//     return low <= arg \&\& arg <= hi;}}
\DoxyCodeLine{1274 \textcolor{comment}{//   \}}}
\DoxyCodeLine{1275 \textcolor{comment}{//   ...}}
\DoxyCodeLine{1276 \textcolor{comment}{//   EXPECT\_THAT(3, InClosedRange(4, 6));}}
\DoxyCodeLine{1277 \textcolor{comment}{//   EXPECT\_THAT(3, Not(InClosedRange(2, 4)));}}
\DoxyCodeLine{1278 \textcolor{comment}{//}}
\DoxyCodeLine{1279 \textcolor{comment}{// would generate two failures that contain the text:}}
\DoxyCodeLine{1280 \textcolor{comment}{//}}
\DoxyCodeLine{1281 \textcolor{comment}{//   Expected: is in range [4, 6]}}
\DoxyCodeLine{1282 \textcolor{comment}{//   ...}}
\DoxyCodeLine{1283 \textcolor{comment}{//   Expected: is not in range [2, 4]}}
\DoxyCodeLine{1284 \textcolor{comment}{//}}
\DoxyCodeLine{1285 \textcolor{comment}{// If you specify "{}"{} as the description, the failure message will}}
\DoxyCodeLine{1286 \textcolor{comment}{// contain the sequence of words in the matcher name followed by the}}
\DoxyCodeLine{1287 \textcolor{comment}{// parameter values printed as a tuple.  For example,}}
\DoxyCodeLine{1288 \textcolor{comment}{//}}
\DoxyCodeLine{1289 \textcolor{comment}{//   MATCHER\_P2(InClosedRange, low, hi, "{}"{}) \{ ... \}}}
\DoxyCodeLine{1290 \textcolor{comment}{//   ...}}
\DoxyCodeLine{1291 \textcolor{comment}{//   EXPECT\_THAT(3, InClosedRange(4, 6));}}
\DoxyCodeLine{1292 \textcolor{comment}{//   EXPECT\_THAT(3, Not(InClosedRange(2, 4)));}}
\DoxyCodeLine{1293 \textcolor{comment}{//}}
\DoxyCodeLine{1294 \textcolor{comment}{// would generate two failures that contain the text:}}
\DoxyCodeLine{1295 \textcolor{comment}{//}}
\DoxyCodeLine{1296 \textcolor{comment}{//   Expected: in closed range (4, 6)}}
\DoxyCodeLine{1297 \textcolor{comment}{//   ...}}
\DoxyCodeLine{1298 \textcolor{comment}{//   Expected: not (in closed range (2, 4))}}
\DoxyCodeLine{1299 \textcolor{comment}{//}}
\DoxyCodeLine{1300 \textcolor{comment}{// Types of Matcher Parameters}}
\DoxyCodeLine{1301 \textcolor{comment}{// ===========================}}
\DoxyCodeLine{1302 \textcolor{comment}{//}}
\DoxyCodeLine{1303 \textcolor{comment}{// For the purpose of typing, you can view}}
\DoxyCodeLine{1304 \textcolor{comment}{//}}
\DoxyCodeLine{1305 \textcolor{comment}{//   MATCHER\_Pk(Foo, p1, ..., pk, description\_string) \{ ... \}}}
\DoxyCodeLine{1306 \textcolor{comment}{//}}
\DoxyCodeLine{1307 \textcolor{comment}{// as shorthand for}}
\DoxyCodeLine{1308 \textcolor{comment}{//}}
\DoxyCodeLine{1309 \textcolor{comment}{//   template <typename p1\_type, ..., typename pk\_type>}}
\DoxyCodeLine{1310 \textcolor{comment}{//   FooMatcherPk<p1\_type, ..., pk\_type>}}
\DoxyCodeLine{1311 \textcolor{comment}{//   Foo(p1\_type p1, ..., pk\_type pk) \{ ... \}}}
\DoxyCodeLine{1312 \textcolor{comment}{//}}
\DoxyCodeLine{1313 \textcolor{comment}{// When you write Foo(v1, ..., vk), the compiler infers the types of}}
\DoxyCodeLine{1314 \textcolor{comment}{// the parameters v1, ..., and vk for you.  If you are not happy with}}
\DoxyCodeLine{1315 \textcolor{comment}{// the result of the type inference, you can specify the types by}}
\DoxyCodeLine{1316 \textcolor{comment}{// explicitly instantiating the template, as in Foo<long, bool>(5,}}
\DoxyCodeLine{1317 \textcolor{comment}{// false).  As said earlier, you don't get to (or need to) specify}}
\DoxyCodeLine{1318 \textcolor{comment}{// 'arg\_type' as that's determined by the context in which the matcher}}
\DoxyCodeLine{1319 \textcolor{comment}{// is used.  You can assign the result of expression Foo(p1, ..., pk)}}
\DoxyCodeLine{1320 \textcolor{comment}{// to a variable of type FooMatcherPk<p1\_type, ..., pk\_type>.  This}}
\DoxyCodeLine{1321 \textcolor{comment}{// can be useful when composing matchers.}}
\DoxyCodeLine{1322 \textcolor{comment}{//}}
\DoxyCodeLine{1323 \textcolor{comment}{// While you can instantiate a matcher template with reference types,}}
\DoxyCodeLine{1324 \textcolor{comment}{// passing the parameters by pointer usually makes your code more}}
\DoxyCodeLine{1325 \textcolor{comment}{// readable.  If, however, you still want to pass a parameter by}}
\DoxyCodeLine{1326 \textcolor{comment}{// reference, be aware that in the failure message generated by the}}
\DoxyCodeLine{1327 \textcolor{comment}{// matcher you will see the value of the referenced object but not its}}
\DoxyCodeLine{1328 \textcolor{comment}{// address.}}
\DoxyCodeLine{1329 \textcolor{comment}{//}}
\DoxyCodeLine{1330 \textcolor{comment}{// Explaining Match Results}}
\DoxyCodeLine{1331 \textcolor{comment}{// ========================}}
\DoxyCodeLine{1332 \textcolor{comment}{//}}
\DoxyCodeLine{1333 \textcolor{comment}{// Sometimes the matcher description alone isn't enough to explain why}}
\DoxyCodeLine{1334 \textcolor{comment}{// the match has failed or succeeded.  For example, when expecting a}}
\DoxyCodeLine{1335 \textcolor{comment}{// long string, it can be very helpful to also print the diff between}}
\DoxyCodeLine{1336 \textcolor{comment}{// the expected string and the actual one.  To achieve that, you can}}
\DoxyCodeLine{1337 \textcolor{comment}{// optionally stream additional information to a special variable}}
\DoxyCodeLine{1338 \textcolor{comment}{// named result\_listener, whose type is a pointer to class}}
\DoxyCodeLine{1339 \textcolor{comment}{// MatchResultListener:}}
\DoxyCodeLine{1340 \textcolor{comment}{//}}
\DoxyCodeLine{1341 \textcolor{comment}{//   MATCHER\_P(EqualsLongString, str, "{}"{}) \{}}
\DoxyCodeLine{1342 \textcolor{comment}{//     if (arg == str) return true;}}
\DoxyCodeLine{1343 \textcolor{comment}{//}}
\DoxyCodeLine{1344 \textcolor{comment}{//     *result\_listener << "{}the difference: "{}}}
\DoxyCodeLine{1346 \textcolor{comment}{}\textcolor{comment}{//     return false;}}
\DoxyCodeLine{1347 \textcolor{comment}{//   \}}}
\DoxyCodeLine{1348 \textcolor{comment}{//}}
\DoxyCodeLine{1349 \textcolor{comment}{// Overloading Matchers}}
\DoxyCodeLine{1350 \textcolor{comment}{// ====================}}
\DoxyCodeLine{1351 \textcolor{comment}{//}}
\DoxyCodeLine{1352 \textcolor{comment}{// You can overload matchers with different numbers of parameters:}}
\DoxyCodeLine{1353 \textcolor{comment}{//}}
\DoxyCodeLine{1354 \textcolor{comment}{//   MATCHER\_P(Blah, a, description\_string1) \{ ... \}}}
\DoxyCodeLine{1355 \textcolor{comment}{//   MATCHER\_P2(Blah, a, b, description\_string2) \{ ... \}}}
\DoxyCodeLine{1356 \textcolor{comment}{//}}
\DoxyCodeLine{1357 \textcolor{comment}{// Caveats}}
\DoxyCodeLine{1358 \textcolor{comment}{// =======}}
\DoxyCodeLine{1359 \textcolor{comment}{//}}
\DoxyCodeLine{1360 \textcolor{comment}{// When defining a new matcher, you should also consider implementing}}
\DoxyCodeLine{1361 \textcolor{comment}{// MatcherInterface or using MakePolymorphicMatcher().  These}}
\DoxyCodeLine{1362 \textcolor{comment}{// approaches require more work than the MATCHER* macros, but also}}
\DoxyCodeLine{1363 \textcolor{comment}{// give you more control on the types of the value being matched and}}
\DoxyCodeLine{1364 \textcolor{comment}{// the matcher parameters, which may leads to better compiler error}}
\DoxyCodeLine{1365 \textcolor{comment}{// messages when the matcher is used wrong.  They also allow}}
\DoxyCodeLine{1366 \textcolor{comment}{// overloading matchers based on parameter types (as opposed to just}}
\DoxyCodeLine{1367 \textcolor{comment}{// based on the number of parameters).}}
\DoxyCodeLine{1368 \textcolor{comment}{//}}
\DoxyCodeLine{1369 \textcolor{comment}{// MATCHER*() can only be used in a namespace scope.  The reason is}}
\DoxyCodeLine{1370 \textcolor{comment}{// that C++ doesn't yet allow function-\/local types to be used to}}
\DoxyCodeLine{1371 \textcolor{comment}{// instantiate templates.  The up-\/coming C++0x standard will fix this.}}
\DoxyCodeLine{1372 \textcolor{comment}{// Once that's done, we'll consider supporting using MATCHER*() inside}}
\DoxyCodeLine{1373 \textcolor{comment}{// a function.}}
\DoxyCodeLine{1374 \textcolor{comment}{//}}
