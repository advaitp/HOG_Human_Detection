21template<typename ExpressionType, int Side, bool Transposed>
22struct permutation_matrix_product<ExpressionType, Side, Transposed, SparseShape>
23{
24 typedef typename nested_eval<ExpressionType, 1>::type MatrixType;
25 typedef typename remove_all<MatrixType>::type MatrixTypeCleaned;
26
27 typedef typename MatrixTypeCleaned::Scalar Scalar;
28 typedef typename MatrixTypeCleaned::StorageIndex StorageIndex;
29
30 enum {
31 SrcStorageOrder = MatrixTypeCleaned::Flags&RowMajorBit ? RowMajor : ColMajor,
32 MoveOuter = SrcStorageOrder==RowMajor ? Side==OnTheLeft : Side==OnTheRight
33 };
34
35 typedef typename internal::conditional<MoveOuter,
36 SparseMatrix<Scalar,SrcStorageOrder,StorageIndex>,
37 SparseMatrix<Scalar,int(SrcStorageOrder)==RowMajor?ColMajor:RowMajor,StorageIndex> >::type ReturnType;
38
39 template<typename Dest,typename PermutationType>
40 static inline void run(Dest& dst, const PermutationType& perm, const ExpressionType& xpr)
41 {
42 MatrixType mat(xpr);
43 if(MoveOuter)
44 {
45 SparseMatrix<Scalar,SrcStorageOrder,StorageIndex> tmp(mat.rows(), mat.cols());
46 Matrix<StorageIndex,Dynamic,1> sizes(mat.outerSize());
47 for(Index j=0; j<mat.outerSize(); ++j)
48 {
49 Index jp = perm.indices().coeff(j);
50 sizes[((Side==OnTheLeft) ^ Transposed) ? jp : j] = StorageIndex(mat.innerVector(((Side==OnTheRight) ^ Transposed) ? jp : j).nonZeros());
51 }
52 tmp.reserve(sizes);
53 for(Index j=0; j<mat.outerSize(); ++j)
54 {
55 Index jp = perm.indices().coeff(j);
56 Index jsrc = ((Side==OnTheRight) ^ Transposed) ? jp : j;
57 Index jdst = ((Side==OnTheLeft) ^ Transposed) ? jp : j;
58 for(typename MatrixTypeCleaned::InnerIterator it(mat,jsrc); it; ++it)
59 tmp.insertByOuterInner(jdst,it.index()) = it.value();
60 }
61 dst = tmp;
62 }
63 else
64 {
65 SparseMatrix<Scalar,int(SrcStorageOrder)==RowMajor?ColMajor:RowMajor,StorageIndex> tmp(mat.rows(), mat.cols());
66 Matrix<StorageIndex,Dynamic,1> sizes(tmp.outerSize());
67 sizes.setZero();
68 PermutationMatrix<Dynamic,Dynamic,StorageIndex> perm_cpy;
69 if((Side==OnTheLeft) ^ Transposed)
70 perm_cpy = perm;
71 else
72 perm_cpy = perm.transpose();
73
74 for(Index j=0; j<mat.outerSize(); ++j)
75 for(typename MatrixTypeCleaned::InnerIterator it(mat,j); it; ++it)
76 sizes[perm_cpy.indices().coeff(it.index())]++;
77 tmp.reserve(sizes);
78 for(Index j=0; j<mat.outerSize(); ++j)
79 for(typename MatrixTypeCleaned::InnerIterator it(mat,j); it; ++it)
80 tmp.insertByOuterInner(perm_cpy.indices().coeff(it.index()),j) = it.value();
81 dst = tmp;
82 }
83 }
84};
