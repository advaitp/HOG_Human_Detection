667template<typename Derived>
668bool MatrixBase<Derived>::isUpperTriangular(const RealScalar& prec) const
669{
670 RealScalar maxAbsOnUpperPart = static_cast<RealScalar>(-1);
671 for(Index j = 0; j < cols(); ++j)
672 {
673 Index maxi = numext::mini(j, rows()-1);
674 for(Index i = 0; i <= maxi; ++i)
675 {
676 RealScalar absValue = numext::abs(coeff(i,j));
677 if(absValue > maxAbsOnUpperPart) maxAbsOnUpperPart = absValue;
678 }
679 }
680 RealScalar threshold = maxAbsOnUpperPart * prec;
681 for(Index j = 0; j < cols(); ++j)
682 for(Index i = j+1; i < rows(); ++i)
683 if(numext::abs(coeff(i, j)) > threshold) return false;
684 return true;
685}
686
692template<typename Derived>
693bool MatrixBase<Derived>::isLowerTriangular(const RealScalar& prec) const
694{
695 RealScalar maxAbsOnLowerPart = static_cast<RealScalar>(-1);
696 for(Index j = 0; j < cols(); ++j)
697 for(Index i = j; i < rows(); ++i)
698 {
699 RealScalar absValue = numext::abs(coeff(i,j));
700 if(absValue > maxAbsOnLowerPart) maxAbsOnLowerPart = absValue;
701 }
702 RealScalar threshold = maxAbsOnLowerPart * prec;
703 for(Index j = 1; j < cols(); ++j)
704 {
705 Index maxi = numext::mini(j, rows()-1);
706 for(Index i = 0; i < maxi; ++i)
707 if(numext::abs(coeff(i, j)) > threshold) return false;
708 }
709 return true;
710}

803template<int Mode, bool SetOpposite, typename DstXprType, typename SrcXprType, typename Functor>
804EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
805void call_triangular_assignment_loop(DstXprType& dst, const SrcXprType& src, const Functor &func)
806{
807 typedef evaluator<DstXprType> DstEvaluatorType;
808 typedef evaluator<SrcXprType> SrcEvaluatorType;
809
810 SrcEvaluatorType srcEvaluator(src);
811
812 Index dstRows = src.rows();
813 Index dstCols = src.cols();
814 if((dst.rows()!=dstRows) || (dst.cols()!=dstCols))
815 dst.resize(dstRows, dstCols);
816 DstEvaluatorType dstEvaluator(dst);
817
818 typedef triangular_dense_assignment_kernel< Mode&(Lower|Upper),Mode&(UnitDiag|ZeroDiag|SelfAdjoint),SetOpposite,
819 DstEvaluatorType,SrcEvaluatorType,Functor> Kernel;
820 Kernel kernel(dstEvaluator, srcEvaluator, func, dst.const_cast_derived());
821
822 enum {
823 unroll = DstXprType::SizeAtCompileTime != Dynamic
824 && SrcEvaluatorType::CoeffReadCost < HugeCost
825 && DstXprType::SizeAtCompileTime * (int(DstEvaluatorType::CoeffReadCost) + int(SrcEvaluatorType::CoeffReadCost)) / 2 <= EIGEN_UNROLLING_LIMIT
826 };

873template<typename Kernel, unsigned int Mode, int UnrollCount, bool SetOpposite>
874struct triangular_assignment_loop
875{
876 // FIXME: this is not very clean, perhaps this information should be provided by the kernel?
877 typedef typename Kernel::DstEvaluatorType DstEvaluatorType;
878 typedef typename DstEvaluatorType::XprType DstXprType;
879
880 enum {
881 col = (UnrollCount-1) / DstXprType::RowsAtCompileTime,
882 row = (UnrollCount-1) % DstXprType::RowsAtCompileTime
883 };
884
885 typedef typename Kernel::Scalar Scalar;
886
887 EIGEN_DEVICE_FUNC
888 static inline void run(Kernel &kernel)
889 {
890 triangular_assignment_loop<Kernel, Mode, UnrollCount-1, SetOpposite>::run(kernel);
891
892 if(row==col)
893 kernel.assignDiagonalCoeff(row);
894 else if( ((Mode&Lower) && row>col) || ((Mode&Upper) && row<col) )
895 kernel.assignCoeff(row,col);
896 else if(SetOpposite)
897 kernel.assignOppositeCoeff(row,col);
898 }
899};

901// prevent buggy user code from causing an infinite recursion
902template<typename Kernel, unsigned int Mode, bool SetOpposite>
903struct triangular_assignment_loop<Kernel, Mode, 0, SetOpposite>
904{
905 EIGEN_DEVICE_FUNC
906 static inline void run(Kernel &) {}
907};

912// triangular part into one rectangular and two triangular parts.

915template<typename Kernel, unsigned int Mode, bool SetOpposite>
916struct triangular_assignment_loop<Kernel, Mode, Dynamic, SetOpposite>
917{
918 typedef typename Kernel::Scalar Scalar;
919 EIGEN_DEVICE_FUNC
920 static inline void run(Kernel &kernel)
921 {
922 for(Index j = 0; j < kernel.cols(); ++j)
923 {
924 Index maxi = numext::mini(j, kernel.rows());
925 Index i = 0;
926 if (((Mode&Lower) && SetOpposite) || (Mode&Upper))
927 {
928 for(; i < maxi; ++i)
929 if(Mode&Upper) kernel.assignCoeff(i, j);
930 else kernel.assignOppositeCoeff(i, j);
931 }
932 else
933 i = maxi;
934
935 if(i<kernel.rows()) // then i==j
936 kernel.assignDiagonalCoeff(i++);
937
938 if (((Mode&Upper) && SetOpposite) || (Mode&Lower))
939 {
940 for(; i < kernel.rows(); ++i)
941 if(Mode&Lower) kernel.assignCoeff(i, j);
942 else kernel.assignOppositeCoeff(i, j);
943 }
944 }
945 }
946};
