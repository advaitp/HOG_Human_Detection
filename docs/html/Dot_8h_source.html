19// helper function for dot(). The problem is that if we put that in the body of dot(), then upon calling dot
20// with mismatched types, the compiler emits errors about failing to instantiate cwiseProduct BEFORE
21// looking at the static assertions. Thus this is a trick to get better compile errors.
22template<typename T, typename U,
23// the NeedToTranspose condition here is taken straight from Assign.h
24 bool NeedToTranspose = T::IsVectorAtCompileTime
25 && U::IsVectorAtCompileTime
26 && ((int(T::RowsAtCompileTime) == 1 && int(U::ColsAtCompileTime) == 1)
27 | // FIXME | instead of || to please GCC 4.4.0 stupid warning "suggest parentheses around &&".
28 // revert to || as soon as not needed anymore.
29 (int(T::ColsAtCompileTime) == 1 && int(U::RowsAtCompileTime) == 1))
30>
31struct dot_nocheck
32{
33 typedef scalar_conj_product_op<typename traits<T>::Scalar,typename traits<U>::Scalar> conj_prod;
34 typedef typename conj_prod::result_type ResScalar;
35 EIGEN_DEVICE_FUNC
36 EIGEN_STRONG_INLINE
37 static ResScalar run(const MatrixBase<T>& a, const MatrixBase<U>& b)
38 {
39 return a.template binaryExpr<conj_prod>(b).sum();
40 }
41};
42
43template<typename T, typename U>
44struct dot_nocheck<T, U, true>
45{
46 typedef scalar_conj_product_op<typename traits<T>::Scalar,typename traits<U>::Scalar> conj_prod;
47 typedef typename conj_prod::result_type ResScalar;
48 EIGEN_DEVICE_FUNC
49 EIGEN_STRONG_INLINE
50 static ResScalar run(const MatrixBase<T>& a, const MatrixBase<U>& b)
51 {
52 return a.transpose().template binaryExpr<conj_prod>(b).sum();
53 }
54};

71EIGEN_DEVICE_FUNC
72EIGEN_STRONG_INLINE
73typename ScalarBinaryOpTraits<typename internal::traits<Derived>::Scalar,typename internal::traits<OtherDerived>::Scalar>::ReturnType
74MatrixBase<Derived>::dot(const MatrixBase<OtherDerived>& other) const
75{
76 EIGEN_STATIC_ASSERT_VECTOR_ONLY(Derived)
77 EIGEN_STATIC_ASSERT_VECTOR_ONLY(OtherDerived)
78 EIGEN_STATIC_ASSERT_SAME_VECTOR_SIZE(Derived,OtherDerived)
79#if !(defined(EIGEN_NO_STATIC_ASSERT) && defined(EIGEN_NO_DEBUG))
80 typedef internal::scalar_conj_product_op<Scalar,typename OtherDerived::Scalar> func;
81 EIGEN_CHECK_BINARY_COMPATIBILIY(func,Scalar,typename OtherDerived::Scalar);
82#endif
83
84 eigen_assert(size() == other.size());
85
86 return internal::dot_nocheck<Derived,OtherDerived>::run(*this, other);
87}

124template<typename Derived>
125EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const typename MatrixBase<Derived>::PlainObject
126MatrixBase<Derived>::normalized() const
127{
128 typedef typename internal::nested_eval<Derived,2>::type _Nested;
129 _Nested n(derived());
130 RealScalar z = n.squaredNorm();
131 // NOTE: after extensive benchmarking, this conditional does not impact performance, at least on recent x86 CPU
132 if(z>RealScalar(0))
133 return n / numext::sqrt(z);
134 else
135 return n;
136}

167template<typename Derived>
168EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const typename MatrixBase<Derived>::PlainObject
169MatrixBase<Derived>::stableNormalized() const
170{
171 typedef typename internal::nested_eval<Derived,3>::type _Nested;
172 _Nested n(derived());
173 RealScalar w = n.cwiseAbs().maxCoeff();
174 RealScalar z = (n/w).squaredNorm();
175 if(z>RealScalar(0))
176 return n / (numext::sqrt(z)*w);
177 else
178 return n;
179}

192template<typename Derived>
193EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void MatrixBase<Derived>::stableNormalize()
194{
195 RealScalar w = cwiseAbs().maxCoeff();
196 RealScalar z = (derived()/w).squaredNorm();
197 if(z>RealScalar(0))
198 derived() /= numext::sqrt(z)*w;
199}

205template<typename Derived, int p>
206struct lpNorm_selector
207{
208 typedef typename NumTraits<typename traits<Derived>::Scalar>::Real RealScalar;
209 EIGEN_DEVICE_FUNC
210 static inline RealScalar run(const MatrixBase<Derived>& m)
211 {
212 EIGEN_USING_STD(pow)
213 return pow(m.cwiseAbs().array().pow(p).sum(), RealScalar(1)/p);
214 }

282template<typename Derived>
283template<typename OtherDerived>
284bool MatrixBase<Derived>::isOrthogonal
285(const MatrixBase<OtherDerived>& other, const RealScalar& prec) const
286{
287 typename internal::nested_eval<Derived,2>::type nested(derived());
288 typename internal::nested_eval<OtherDerived,2>::type otherNested(other.derived());
289 return numext::abs2(nested.dot(otherNested)) <= prec * prec * nested.squaredNorm() * otherNested.squaredNorm();
290}

303template<typename Derived>
304bool MatrixBase<Derived>::isUnitary(const RealScalar& prec) const
305{
306 typename internal::nested_eval<Derived,1>::type self(derived());
307 for(Index i = 0; i < cols(); ++i)
308 {
309 if(!internal::isApprox(self.col(i).squaredNorm(), static_cast<RealScalar>(1), prec))
310 return false;
311 for(Index j = 0; j < i; ++j)
312 if(!internal::isMuchSmallerThan(self.col(i).dot(self.col(j)), static_cast<Scalar>(1), prec))
313 return false;
314 }
315 return true;
316}
