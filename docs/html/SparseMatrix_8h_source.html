192 inline Scalar coeff(Index row, Index col) const
193 {
194 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
195
196 const Index outer = IsRowMajor ? row : col;
197 const Index inner = IsRowMajor ? col : row;
198 Index end = m_innerNonZeros ? m_outerIndex[outer] + m_innerNonZeros[outer] : m_outerIndex[outer+1];
199 return m_data.atInRange(m_outerIndex[outer], end, StorageIndex(inner));

210 inline Scalar& coeffRef(Index row, Index col)
211 {
212 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
213
214 const Index outer = IsRowMajor ? row : col;
215 const Index inner = IsRowMajor ? col : row;
216
217 Index start = m_outerIndex[outer];
218 Index end = m_innerNonZeros ? m_outerIndex[outer] + m_innerNonZeros[outer] : m_outerIndex[outer+1];
219 eigen_assert(end>=start && "you probably called coeffRef on a non finalized matrix");
220 if(end<=start)
221 return insert(row,col);
222 const Index p = m_data.searchLowerIndex(start,end-1,StorageIndex(inner));
223 if((p<end) && (m_data.index(p)==inner))
224 return m_data.value(p);
225 else
226 return insert(row,col);
227 }

301 template<class SizesType>
302 inline void reserveInnerVectors(const SizesType& reserveSizes)
303 {
304 if(isCompressed())
305 {
306 Index totalReserveSize = 0;
307 // turn the matrix into non-compressed mode
308 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
309 if (!m_innerNonZeros) internal::throw_std_bad_alloc();
310
311 // temporarily use m_innerSizes to hold the new starting points.
312 StorageIndex* newOuterIndex = m_innerNonZeros;
313
314 StorageIndex count = 0;
315 for(Index j=0; j<m_outerSize; ++j)
316 {
317 newOuterIndex[j] = count;
318 count += reserveSizes[j] + (m_outerIndex[j+1]-m_outerIndex[j]);
319 totalReserveSize += reserveSizes[j];
320 }
321 m_data.reserve(totalReserveSize);
322 StorageIndex previousOuterIndex = m_outerIndex[m_outerSize];
323 for(Index j=m_outerSize-1; j>=0; --j)
324 {
325 StorageIndex innerNNZ = previousOuterIndex - m_outerIndex[j];
326 for(Index i=innerNNZ-1; i>=0; --i)
327 {
328 m_data.index(newOuterIndex[j]+i) = m_data.index(m_outerIndex[j]+i);
329 m_data.value(newOuterIndex[j]+i) = m_data.value(m_outerIndex[j]+i);
330 }
331 previousOuterIndex = m_outerIndex[j];
332 m_outerIndex[j] = newOuterIndex[j];
333 m_innerNonZeros[j] = innerNNZ;
334 }
335 if(m_outerSize>0)
336 m_outerIndex[m_outerSize] = m_outerIndex[m_outerSize-1] + m_innerNonZeros[m_outerSize-1] + reserveSizes[m_outerSize-1];
337
338 m_data.resize(m_outerIndex[m_outerSize]);
339 }
340 else
341 {
342 StorageIndex* newOuterIndex = static_cast<StorageIndex*>(std::malloc((m_outerSize+1)*sizeof(StorageIndex)));
343 if (!newOuterIndex) internal::throw_std_bad_alloc();
344
345 StorageIndex count = 0;
346 for(Index j=0; j<m_outerSize; ++j)
347 {
348 newOuterIndex[j] = count;
349 StorageIndex alreadyReserved = (m_outerIndex[j+1]-m_outerIndex[j]) - m_innerNonZeros[j];
350 StorageIndex toReserve = std::max<StorageIndex>(reserveSizes[j], alreadyReserved);
351 count += toReserve + m_innerNonZeros[j];
352 }
353 newOuterIndex[m_outerSize] = count;
354
355 m_data.resize(count);
356 for(Index j=m_outerSize-1; j>=0; --j)
357 {
358 Index offset = newOuterIndex[j] - m_outerIndex[j];
359 if(offset>0)
360 {
361 StorageIndex innerNNZ = m_innerNonZeros[j];
362 for(Index i=innerNNZ-1; i>=0; --i)
363 {
364 m_data.index(newOuterIndex[j]+i) = m_data.index(m_outerIndex[j]+i);
365 m_data.value(newOuterIndex[j]+i) = m_data.value(m_outerIndex[j]+i);
366 }
367 }
368 }
369
370 std::swap(m_outerIndex, newOuterIndex);
371 std::free(newOuterIndex);
372 }

428 inline void finalize()
429 {
430 if(isCompressed())
431 {
432 StorageIndex size = internal::convert_index<StorageIndex>(m_data.size());
433 Index i = m_outerSize;
434 // find the last filled column
435 while (i>=0 && m_outerIndex[i]==0)
436 --i;
437 ++i;
438 while (i<=m_outerSize)

440 m_outerIndex[i] = size;
441 ++i;
442 }

470 void makeCompressed()
471 {
472 if(isCompressed())
473 return;
474
475 eigen_internal_assert(m_outerIndex!=0 && m_outerSize>0);
476
477 Index oldStart = m_outerIndex[1];
478 m_outerIndex[1] = m_innerNonZeros[0];
479 for(Index j=1; j<m_outerSize; ++j)
480 {
481 Index nextOldStart = m_outerIndex[j+1];
482 Index offset = oldStart - m_outerIndex[j];
483 if(offset>0)
484 {
485 for(Index k=0; k<m_innerNonZeros[j]; ++k)
486 {
487 m_data.index(m_outerIndex[j]+k) = m_data.index(oldStart+k);
488 m_data.value(m_outerIndex[j]+k) = m_data.value(oldStart+k);
489 }
490 }
491 m_outerIndex[j+1] = m_outerIndex[j] + m_innerNonZeros[j];
492 oldStart = nextOldStart;
493 }
494 std::free(m_innerNonZeros);
495 m_innerNonZeros = 0;
496 m_data.resize(m_outerIndex[m_outerSize]);
497 m_data.squeeze();
498 }

501 void uncompress()
502 {
503 if(m_innerNonZeros != 0)
504 return;

507 {
508 m_innerNonZeros[i] = m_outerIndex[i+1] - m_outerIndex[i];

525 template<typename KeepFunc>
526 void prune(const KeepFunc& keep = KeepFunc())
527 {
528 // TODO optimize the uncompressed mode to avoid moving and allocating the data twice
529 makeCompressed();
530
531 StorageIndex k = 0;
532 for(Index j=0; j<m_outerSize; ++j)
533 {
534 Index previousStart = m_outerIndex[j];
535 m_outerIndex[j] = k;
536 Index end = m_outerIndex[j+1];
537 for(Index i=previousStart; i<end; ++i)
538 {
539 if(keep(IsRowMajor?j:m_data.index(i), IsRowMajor?m_data.index(i):j, m_data.value(i)))
540 {
541 m_data.value(k) = m_data.value(i);
542 m_data.index(k) = m_data.index(i);
543 ++k;
544 }
545 }
546 }
547 m_outerIndex[m_outerSize] = k;
548 m_data.resize(k,0);
549 }

559 void conservativeResize(Index rows, Index cols)
560 {
561 // No change
562 if (this->rows() == rows && this->cols() == cols) return;
563
564 // If one dimension is null, then there is nothing to be preserved
565 if(rows==0 || cols==0) return resize(rows,cols);
566
567 Index innerChange = IsRowMajor ? cols - this->cols() : rows - this->rows();
568 Index outerChange = IsRowMajor ? rows - this->rows() : cols - this->cols();
569 StorageIndex newInnerSize = convert_index(IsRowMajor ? cols : rows);
570
571 // Deals with inner non zeros
572 if (m_innerNonZeros)
573 {
574 // Resize m_innerNonZeros
575 StorageIndex *newInnerNonZeros = static_cast<StorageIndex*>(std::realloc(m_innerNonZeros, (m_outerSize + outerChange) * sizeof(StorageIndex)));
576 if (!newInnerNonZeros) internal::throw_std_bad_alloc();
577 m_innerNonZeros = newInnerNonZeros;
578
579 for(Index i=m_outerSize; i<m_outerSize+outerChange; i++)
580 m_innerNonZeros[i] = 0;
581 }
582 else if (innerChange < 0)
583 {
584 // Inner size decreased: allocate a new m_innerNonZeros
585 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc((m_outerSize + outerChange) * sizeof(StorageIndex)));
586 if (!m_innerNonZeros) internal::throw_std_bad_alloc();
587 for(Index i = 0; i < m_outerSize + (std::min)(outerChange, Index(0)); i++)
588 m_innerNonZeros[i] = m_outerIndex[i+1] - m_outerIndex[i];
589 for(Index i = m_outerSize; i < m_outerSize + outerChange; i++)
590 m_innerNonZeros[i] = 0;
591 }
592
593 // Change the m_innerNonZeros in case of a decrease of inner size
594 if (m_innerNonZeros && innerChange < 0)
595 {
596 for(Index i = 0; i < m_outerSize + (std::min)(outerChange, Index(0)); i++)
597 {
598 StorageIndex &n = m_innerNonZeros[i];
599 StorageIndex start = m_outerIndex[i];
600 while (n > 0 && m_data.index(start+n-1) >= newInnerSize) --n;
601 }
602 }
603
604 m_innerSize = newInnerSize;
605
606 // Re-allocate outer index structure if necessary
607 if (outerChange == 0)
608 return;
609
610 StorageIndex *newOuterIndex = static_cast<StorageIndex*>(std::realloc(m_outerIndex, (m_outerSize + outerChange + 1) * sizeof(StorageIndex)));
611 if (!newOuterIndex) internal::throw_std_bad_alloc();
612 m_outerIndex = newOuterIndex;
613 if (outerChange > 0)
614 {
615 StorageIndex lastIdx = m_outerSize == 0 ? 0 : m_outerIndex[m_outerSize];
616 for(Index i=m_outerSize; i<m_outerSize+outerChange+1; i++)
617 m_outerIndex[i] = lastIdx;
618 }
619 m_outerSize += outerChange;
620 }

629 void resize(Index rows, Index cols)
630 {
631 const Index outerSize = IsRowMajor ? rows : cols;
632 m_innerSize = IsRowMajor ? cols : rows;
633 m_data.clear();
634 if (m_outerSize != outerSize || m_outerSize==0)
635 {
636 std::free(m_outerIndex);
637 m_outerIndex = static_cast<StorageIndex*>(std::malloc((outerSize + 1) * sizeof(StorageIndex)));
638 if (!m_outerIndex) internal::throw_std_bad_alloc();
639
640 m_outerSize = outerSize;
641 }
642 if(m_innerNonZeros)
643 {
644 std::free(m_innerNonZeros);
645 m_innerNonZeros = 0;
646 }
647 std::fill_n(m_outerIndex, m_outerSize + 1, StorageIndex(0));
648 }

745 inline void setIdentity()
746 {
747 eigen_assert(rows() == cols() && "ONLY FOR SQUARED MATRICES");
748 this->m_data.resize(rows());
749 Eigen::Map<IndexVector>(this->m_data.indexPtr(), rows()).setLinSpaced(0, StorageIndex(rows()-1));

756 {
757 if (other.isRValue())

762 {
763 #ifdef EIGEN_SPARSE_CREATE_TEMPORARY_PLUGIN

766 initAssignment(other);

770 m_data = other.m_data;
771 }
772 else
773 {
774 Base::operator=(other);
775 }
776 }
777 return *this;

922 template<typename DiagXpr, typename Func>
923 void assignDiagonal(const DiagXpr diagXpr, const Func& assignFunc)
924 {
925 Index n = diagXpr.size();
926
927 const bool overwrite = internal::is_same<Func, internal::assign_op<Scalar,Scalar> >::value;
928 if(overwrite)
929 {
930 if((this->rows()!=n) || (this->cols()!=n))
931 this->resize(n, n);
932 }
933
934 if(m_data.size()==0 || overwrite)
935 {
936 typedef Array<StorageIndex,Dynamic,1> ArrayXI;
937 this->makeCompressed();
938 this->resizeNonZeros(n);
939 Eigen::Map<ArrayXI>(this->innerIndexPtr(), n).setLinSpaced(0,StorageIndex(n)-1);
940 Eigen::Map<ArrayXI>(this->outerIndexPtr(), n+1).setLinSpaced(0,StorageIndex(n));
941 Eigen::Map<Array<Scalar,Dynamic,1> > values = this->coeffs();
942 values.setZero();
943 internal::call_assignment_no_alias(values, diagXpr, assignFunc);
944 }
945 else
946 {
947 bool isComp = isCompressed();
948 internal::evaluator<DiagXpr> diaEval(diagXpr);
949 std::vector<IndexPosPair> newEntries;
950
951 // 1 - try in-place update and record insertion failures
952 for(Index i = 0; i<n; ++i)
953 {
954 internal::LowerBoundIndex lb = this->lower_bound(i,i);
955 Index p = lb.value;
956 if(lb.found)
957 {
958 // the coeff already exists
959 assignFunc.assignCoeff(m_data.value(p), diaEval.coeff(i));
960 }
961 else if((!isComp) && m_innerNonZeros[i] < (m_outerIndex[i+1]-m_outerIndex[i]))
962 {
963 // non compressed mode with local room for inserting one element
964 m_data.moveChunk(p, p+1, m_outerIndex[i]+m_innerNonZeros[i]-p);
965 m_innerNonZeros[i]++;
966 m_data.value(p) = Scalar(0);
967 m_data.index(p) = StorageIndex(i);
968 assignFunc.assignCoeff(m_data.value(p), diaEval.coeff(i));
969 }
970 else
971 {
972 // defer insertion
973 newEntries.push_back(IndexPosPair(i,p));
974 }
975 }
976 // 2 - insert deferred entries
977 Index n_entries = Index(newEntries.size());
978 if(n_entries>0)
979 {
980 Storage newData(m_data.size()+n_entries);
981 Index prev_p = 0;
982 Index prev_i = 0;
983 for(Index k=0; k<n_entries;++k)
984 {
985 Index i = newEntries[k].i;
986 Index p = newEntries[k].p;
987 internal::smart_copy(m_data.valuePtr()+prev_p, m_data.valuePtr()+p, newData.valuePtr()+prev_p+k);
988 internal::smart_copy(m_data.indexPtr()+prev_p, m_data.indexPtr()+p, newData.indexPtr()+prev_p+k);
989 for(Index j=prev_i;j<i;++j)
990 m_outerIndex[j+1] += k;
991 if(!isComp)
992 m_innerNonZeros[i]++;
993 prev_p = p;
994 prev_i = i;
995 newData.value(p+k) = Scalar(0);
996 newData.index(p+k) = StorageIndex(i);
997 assignFunc.assignCoeff(newData.value(p+k), diaEval.coeff(i));
998 }
999 {
1000 internal::smart_copy(m_data.valuePtr()+prev_p, m_data.valuePtr()+m_data.size(), newData.valuePtr()+prev_p+n_entries);
1001 internal::smart_copy(m_data.indexPtr()+prev_p, m_data.indexPtr()+m_data.size(), newData.indexPtr()+prev_p+n_entries);
1002 for(Index j=prev_i+1;j<=m_outerSize;++j)
1003 m_outerIndex[j] += n_entries;
1004 }
1005 m_data.swap(newData);
1006 }
1007 }
1008 }

1027template<typename InputIterator, typename SparseMatrixType, typename DupFunctor>
1028void set_from_triplets(const InputIterator& begin, const InputIterator& end, SparseMatrixType& mat, DupFunctor dup_func)
1029{
1030 enum { IsRowMajor = SparseMatrixType::IsRowMajor };
1031 typedef typename SparseMatrixType::Scalar Scalar;
1032 typedef typename SparseMatrixType::StorageIndex StorageIndex;
1033 SparseMatrix<Scalar,IsRowMajor?ColMajor:RowMajor,StorageIndex> trMat(mat.rows(),mat.cols());
1034
1035 if(begin!=end)
1036 {
1037 // pass 1: count the nnz per inner-vector
1038 typename SparseMatrixType::IndexVector wi(trMat.outerSize());
1039 wi.setZero();
1040 for(InputIterator it(begin); it!=end; ++it)
1041 {
1042 eigen_assert(it->row()>=0 && it->row()<mat.rows() && it->col()>=0 && it->col()<mat.cols());
1043 wi(IsRowMajor ? it->col() : it->row())++;
1044 }
1045
1046 // pass 2: insert all the elements into trMat
1047 trMat.reserve(wi);
1048 for(InputIterator it(begin); it!=end; ++it)
1049 trMat.insertBackUncompressed(it->row(),it->col()) = it->value();
1050
1051 // pass 3:
1052 trMat.collapseDuplicates(dup_func);
1053 }
1054
1055 // pass 4: transposed copy -> implicit sorting
1056 mat = trMat;
1057}
1058

1123template<typename Scalar, int Options_, typename StorageIndex_>
1124template<typename DupFunctor>
1125void SparseMatrix<Scalar,Options_,StorageIndex_>::collapseDuplicates(DupFunctor dup_func)
1126{
1127 eigen_assert(!isCompressed());
1128 // TODO, in practice we should be able to use m_innerNonZeros for that task
1129 IndexVector wi(innerSize());
1130 wi.fill(-1);
1131 StorageIndex count = 0;
1132 // for each inner-vector, wi[inner_index] will hold the position of first element into the index/value buffers
1133 for(Index j=0; j<outerSize(); ++j)
1134 {
1135 StorageIndex start = count;
1136 Index oldEnd = m_outerIndex[j]+m_innerNonZeros[j];
1137 for(Index k=m_outerIndex[j]; k<oldEnd; ++k)
1138 {
1139 Index i = m_data.index(k);
1140 if(wi(i)>=start)
1141 {
1142 // we already meet this entry => accumulate it
1143 m_data.value(wi(i)) = dup_func(m_data.value(wi(i)), m_data.value(k));
1144 }
1145 else
1146 {
1147 m_data.value(count) = m_data.value(k);
1148 m_data.index(count) = m_data.index(k);
1149 wi(i) = count;
1150 ++count;
1151 }
1152 }
1153 m_outerIndex[j] = start;
1154 }
1155 m_outerIndex[m_outerSize] = count;
1156
1157 // turn the matrix into compressed form
1158 std::free(m_innerNonZeros);
1159 m_innerNonZeros = 0;
1160 m_data.resize(m_outerIndex[m_outerSize]);
1161}

1168 YOU_MIXED_DIFFERENT_NUMERIC_TYPES__YOU_NEED_TO_USE_THE_CAST_METHOD_OF_MATRIXBASE_TO_CAST_NUMERIC_TYPES_EXPLICITLY)

1171 EIGEN_SPARSE_CREATE_TEMPORARY_PLUGIN

1175 if (needToTranspose)
1176 {
1177 #ifdef EIGEN_SPARSE_TRANSPOSED_COPY_PLUGIN
1178 EIGEN_SPARSE_TRANSPOSED_COPY_PLUGIN
1179 #endif
1180 // two passes algorithm:
1181 // 1 - compute the number of coeffs per dest inner vector
1182 // 2 - do the actual copy/eval

1184 typedef typename internal::nested_eval<OtherDerived,2,typename internal::plain_matrix_type<OtherDerived>::type >::type OtherCopy;
1185 typedef typename internal::remove_all<OtherCopy>::type _OtherCopy;
1186 typedef internal::evaluator<_OtherCopy> OtherCopyEval;
1187 OtherCopy otherCopy(other.derived());
1188 OtherCopyEval otherCopyEval(otherCopy);
1189
1190 SparseMatrix dest(other.rows(),other.cols());
1191 Eigen::Map<IndexVector> (dest.m_outerIndex,dest.outerSize()).setZero();
1192
1193 // pass 1
1194 // FIXME the above copy could be merged with that pass
1195 for (Index j=0; j<otherCopy.outerSize(); ++j)

1197 ++dest.m_outerIndex[it.index()];
1198
1199 // prefix sum
1200 StorageIndex count = 0;
1201 IndexVector positions(dest.outerSize());
1202 for (Index j=0; j<dest.outerSize(); ++j)
1203 {
1204 StorageIndex tmp = dest.m_outerIndex[j];
1205 dest.m_outerIndex[j] = count;
1206 positions[j] = count;

1208 }
1209 dest.m_outerIndex[dest.outerSize()] = count;

1214 {
1215 for (typename OtherCopyEval::InnerIterator it(otherCopyEval, j); it; ++it)
1216 {
1217 Index pos = positions[it.index()]++;
1218 dest.m_data.index(pos) = j;
1219 dest.m_data.value(pos) = it.value();
1220 }
1221 }
1222 this->swap(dest);

1224 }
1225 else
1226 {

1229 initAssignment(other.derived());

1236template<typename Scalar_, int Options_, typename StorageIndex_>
1237typename SparseMatrix<Scalar_,Options_,StorageIndex_>::Scalar& SparseMatrix<Scalar_,Options_,StorageIndex_>::insert(Index row, Index col)
1238{
1239 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
1240
1241 const Index outer = IsRowMajor ? row : col;
1242 const Index inner = IsRowMajor ? col : row;
1243
1244 if(isCompressed())
1245 {
1246 if(nonZeros()==0)
1247 {
1248 // reserve space if not already done
1249 if(m_data.allocatedSize()==0)
1250 m_data.reserve(2*m_innerSize);
1251
1252 // turn the matrix into non-compressed mode
1253 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
1254 if(!m_innerNonZeros) internal::throw_std_bad_alloc();
1255
1256 std::fill(m_innerNonZeros, m_innerNonZeros + m_outerSize, StorageIndex(0));
1257
1258 // pack all inner-vectors to the end of the pre-allocated space
1259 // and allocate the entire free-space to the first inner-vector
1260 StorageIndex end = convert_index(m_data.allocatedSize());
1261 for(Index j=1; j<=m_outerSize; ++j)
1262 m_outerIndex[j] = end;
1263 }
1264 else
1265 {
1266 // turn the matrix into non-compressed mode
1267 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
1268 if(!m_innerNonZeros) internal::throw_std_bad_alloc();
1269 for(Index j=0; j<m_outerSize; ++j)
1270 m_innerNonZeros[j] = m_outerIndex[j+1]-m_outerIndex[j];
1271 }
1272 }
1273
1274 // check whether we can do a fast "push back" insertion
1275 Index data_end = m_data.allocatedSize();
1276
1277 // First case: we are filling a new inner vector which is packed at the end.
1278 // We assume that all remaining inner-vectors are also empty and packed to the end.
1279 if(m_outerIndex[outer]==data_end)
1280 {
1281 eigen_internal_assert(m_innerNonZeros[outer]==0);
1282
1283 // pack previous empty inner-vectors to end of the used-space
1284 // and allocate the entire free-space to the current inner-vector.
1285 StorageIndex p = convert_index(m_data.size());
1286 Index j = outer;
1287 while(j>=0 && m_innerNonZeros[j]==0)
1288 m_outerIndex[j--] = p;
1289
1290 // push back the new element
1291 ++m_innerNonZeros[outer];
1292 m_data.append(Scalar(0), inner);
1293
1294 // check for reallocation
1295 if(data_end != m_data.allocatedSize())
1296 {
1297 // m_data has been reallocated
1298 // -> move remaining inner-vectors back to the end of the free-space
1299 // so that the entire free-space is allocated to the current inner-vector.
1300 eigen_internal_assert(data_end < m_data.allocatedSize());
1301 StorageIndex new_end = convert_index(m_data.allocatedSize());
1302 for(Index k=outer+1; k<=m_outerSize; ++k)
1303 if(m_outerIndex[k]==data_end)
1304 m_outerIndex[k] = new_end;
1305 }
1306 return m_data.value(p);
1307 }
1308
1309 // Second case: the next inner-vector is packed to the end
1310 // and the current inner-vector end match the used-space.
1311 if(m_outerIndex[outer+1]==data_end && m_outerIndex[outer]+m_innerNonZeros[outer]==m_data.size())
1312 {
1313 eigen_internal_assert(outer+1==m_outerSize || m_innerNonZeros[outer+1]==0);
1314
1315 // add space for the new element
1316 ++m_innerNonZeros[outer];
1317 m_data.resize(m_data.size()+1);
1318
1319 // check for reallocation
1320 if(data_end != m_data.allocatedSize())
1321 {
1322 // m_data has been reallocated
1323 // -> move remaining inner-vectors back to the end of the free-space
1324 // so that the entire free-space is allocated to the current inner-vector.
1325 eigen_internal_assert(data_end < m_data.allocatedSize());
1326 StorageIndex new_end = convert_index(m_data.allocatedSize());
1327 for(Index k=outer+1; k<=m_outerSize; ++k)
1328 if(m_outerIndex[k]==data_end)
1329 m_outerIndex[k] = new_end;
1330 }
1331
1332 // and insert it at the right position (sorted insertion)
1333 Index startId = m_outerIndex[outer];
1334 Index p = m_outerIndex[outer]+m_innerNonZeros[outer]-1;
1335 while ( (p > startId) && (m_data.index(p-1) > inner) )
1336 {
1337 m_data.index(p) = m_data.index(p-1);
1338 m_data.value(p) = m_data.value(p-1);
1339 --p;
1340 }
1341
1342 m_data.index(p) = convert_index(inner);
1343 return (m_data.value(p) = Scalar(0));
1344 }
1345
1346 if(m_data.size() != m_data.allocatedSize())
1347 {
1348 // make sure the matrix is compatible to random un-compressed insertion:
1349 m_data.resize(m_data.allocatedSize());
1350 this->reserveInnerVectors(Array<StorageIndex,Dynamic,1>::Constant(m_outerSize, 2));
1351 }
1352
1353 return insertUncompressed(row,col);
1354}

1356template<typename Scalar_, int Options_, typename StorageIndex_>
1357EIGEN_DONT_INLINE typename SparseMatrix<Scalar_,Options_,StorageIndex_>::Scalar& SparseMatrix<Scalar_,Options_,StorageIndex_>::insertUncompressed(Index row, Index col)
1358{
1359 eigen_assert(!isCompressed());
1360
1361 const Index outer = IsRowMajor ? row : col;
1362 const StorageIndex inner = convert_index(IsRowMajor ? col : row);
1363
1364 Index room = m_outerIndex[outer+1] - m_outerIndex[outer];
1365 StorageIndex innerNNZ = m_innerNonZeros[outer];
1366 if(innerNNZ>=room)
1367 {
1368 // this inner vector is full, we need to reallocate the whole buffer :(
1369 reserve(SingletonVector(outer,std::max<StorageIndex>(2,innerNNZ)));
1370 }
1371
1372 Index startId = m_outerIndex[outer];
1373 Index p = startId + m_innerNonZeros[outer];
1374 while ( (p > startId) && (m_data.index(p-1) > inner) )
1375 {
1376 m_data.index(p) = m_data.index(p-1);
1377 m_data.value(p) = m_data.value(p-1);
1378 --p;
1379 }
1380 eigen_assert((p<=startId || m_data.index(p-1)!=inner) && "you cannot insert an element that already exists, you must call coeffRef to this end");
1381
1382 m_innerNonZeros[outer]++;
1383
1384 m_data.index(p) = inner;
1385 return (m_data.value(p) = Scalar(0));
1386}

1388template<typename Scalar_, int Options_, typename StorageIndex_>
1389EIGEN_DONT_INLINE typename SparseMatrix<Scalar_,Options_,StorageIndex_>::Scalar& SparseMatrix<Scalar_,Options_,StorageIndex_>::insertCompressed(Index row, Index col)
1390{
1391 eigen_assert(isCompressed());
1392
1393 const Index outer = IsRowMajor ? row : col;
1394 const Index inner = IsRowMajor ? col : row;
1395
1396 Index previousOuter = outer;
1397 if (m_outerIndex[outer+1]==0)
1398 {
1399 // we start a new inner vector
1400 while (previousOuter>=0 && m_outerIndex[previousOuter]==0)
1401 {
1402 m_outerIndex[previousOuter] = convert_index(m_data.size());
1403 --previousOuter;
1404 }
1405 m_outerIndex[outer+1] = m_outerIndex[outer];
1406 }
1407
1408 // here we have to handle the tricky case where the outerIndex array
1409 // starts with: [ 0 0 0 0 0 1 ...] and we are inserted in, e.g.,
1410 // the 2nd inner vector...
1411 bool isLastVec = (!(previousOuter==-1 && m_data.size()!=0))
1412 && (std::size_t(m_outerIndex[outer+1]) == m_data.size());
1413
1414 std::size_t startId = m_outerIndex[outer];
1415 // FIXME let's make sure sizeof(long int) == sizeof(std::size_t)
1416 std::size_t p = m_outerIndex[outer+1];
1417 ++m_outerIndex[outer+1];
1418
1419 double reallocRatio = 1;
1420 if (m_data.allocatedSize()<=m_data.size())
1421 {
1422 // if there is no preallocated memory, let's reserve a minimum of 32 elements
1423 if (m_data.size()==0)
1424 {
1425 m_data.reserve(32);
1426 }
1427 else
1428 {
1429 // we need to reallocate the data, to reduce multiple reallocations
1430 // we use a smart resize algorithm based on the current filling ratio
1431 // in addition, we use double to avoid integers overflows
1432 double nnzEstimate = double(m_outerIndex[outer])*double(m_outerSize)/double(outer+1);
1433 reallocRatio = (nnzEstimate-double(m_data.size()))/double(m_data.size());
1434 // furthermore we bound the realloc ratio to:
1435 // 1) reduce multiple minor realloc when the matrix is almost filled
1436 // 2) avoid to allocate too much memory when the matrix is almost empty
1437 reallocRatio = (std::min)((std::max)(reallocRatio,1.5),8.);
1438 }
1439 }
1440 m_data.resize(m_data.size()+1,reallocRatio);
1441
1442 if (!isLastVec)
1443 {
1444 if (previousOuter==-1)
1445 {
1446 // oops wrong guess.
1447 // let's correct the outer offsets
1448 for (Index k=0; k<=(outer+1); ++k)
1449 m_outerIndex[k] = 0;
1450 Index k=outer+1;
1451 while(m_outerIndex[k]==0)
1452 m_outerIndex[k++] = 1;
1453 while (k<=m_outerSize && m_outerIndex[k]!=0)
1454 m_outerIndex[k++]++;
1455 p = 0;
1456 --k;
1457 k = m_outerIndex[k]-1;
1458 while (k>0)
1459 {
1460 m_data.index(k) = m_data.index(k-1);
1461 m_data.value(k) = m_data.value(k-1);
1462 k--;
1463 }
1464 }
1465 else
1466 {
1467 // we are not inserting into the last inner vec
1468 // update outer indices:
1469 Index j = outer+2;
1470 while (j<=m_outerSize && m_outerIndex[j]!=0)
1471 m_outerIndex[j++]++;
1472 --j;
1473 // shift data of last vecs:
1474 Index k = m_outerIndex[j]-1;
1475 while (k>=Index(p))
1476 {
1477 m_data.index(k) = m_data.index(k-1);
1478 m_data.value(k) = m_data.value(k-1);
1479 k--;
1480 }
1481 }
1482 }
1483
1484 while ( (p > startId) && (m_data.index(p-1) > inner) )
1485 {
1486 m_data.index(p) = m_data.index(p-1);
1487 m_data.value(p) = m_data.value(p-1);
1488 --p;
1489 }
1490
1491 m_data.index(p) = inner;
1492 return (m_data.value(p) = Scalar(0));
1493}
